    fedora-packager \

Python Example

{% github_code octocat/Hello-World/blob/master/app.py %}
app.py 
#!/usr/bin/env python3
"""Hello World Application"""

def greet(name: str) -> str:
    """Return a greeting message."""
    return f"Hello, {name}!"

def main():
    message = greet("World")
    print(message)

if __name__ == "__main__":
    main()
